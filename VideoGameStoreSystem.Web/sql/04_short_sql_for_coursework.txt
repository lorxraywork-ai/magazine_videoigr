-- Краткая версия SQL-скрипта для курсовой работы
-- Содержит только основные объекты БД магазина видеоигр

CREATE DATABASE [magazine_videoigr];
GO

USE [magazine_videoigr];
GO

CREATE TABLE [APP_ROLE]
(
    [ROLE_ID] int IDENTITY(1,1) PRIMARY KEY,
    [ROLE_NAME] nvarchar(100) NOT NULL UNIQUE
);
GO

CREATE TABLE [APP_USER]
(
    [USER_ID] int IDENTITY(1,1) PRIMARY KEY,
    [LOGIN] nvarchar(50) NOT NULL UNIQUE,
    [PASSWORD_HASH] nvarchar(512) NOT NULL,
    [FULL_NAME] nvarchar(150) NOT NULL,
    [EMAIL] nvarchar(150) NOT NULL,
    [ROLE_ID] int NOT NULL,
    [IS_ACTIVE] bit NOT NULL DEFAULT 1,
    [CREATED_AT] datetime2 NOT NULL DEFAULT SYSDATETIME(),
    CONSTRAINT [FK_APP_USER_APP_ROLE]
        FOREIGN KEY ([ROLE_ID]) REFERENCES [APP_ROLE]([ROLE_ID])
);
GO

CREATE TABLE [CATEGORY]
(
    [CATEGORY_ID] int IDENTITY(1,1) PRIMARY KEY,
    [CATEGORY_NAME] nvarchar(100) NOT NULL UNIQUE
);
GO

CREATE TABLE [PRODUCT]
(
    [PRODUCT_ID] int IDENTITY(1,1) PRIMARY KEY,
    [PRODUCT_NAME] nvarchar(200) NOT NULL,
    [CATEGORY_ID] int NOT NULL,
    [SKU] nvarchar(40) NOT NULL UNIQUE,
    [DESCRIPTION] nvarchar(2000) NULL,
    [PRICE] decimal(18,2) NOT NULL,
    [COST_PRICE] decimal(18,2) NOT NULL,
    [STOCK_QUANTITY] int NOT NULL DEFAULT 0,
    [IS_ACTIVE] bit NOT NULL DEFAULT 1,
    CONSTRAINT [FK_PRODUCT_CATEGORY]
        FOREIGN KEY ([CATEGORY_ID]) REFERENCES [CATEGORY]([CATEGORY_ID])
);
GO

CREATE TABLE [SUPPLIER]
(
    [SUPPLIER_ID] int IDENTITY(1,1) PRIMARY KEY,
    [SUPPLIER_NAME] nvarchar(150) NOT NULL,
    [PHONE] nvarchar(50) NULL,
    [EMAIL] nvarchar(150) NULL
);
GO

CREATE TABLE [CUSTOMER]
(
    [CUSTOMER_ID] int IDENTITY(1,1) PRIMARY KEY,
    [FULL_NAME] nvarchar(150) NOT NULL,
    [PHONE] nvarchar(50) NULL,
    [EMAIL] nvarchar(150) NULL,
    [USER_ID] int NULL,
    CONSTRAINT [FK_CUSTOMER_APP_USER]
        FOREIGN KEY ([USER_ID]) REFERENCES [APP_USER]([USER_ID])
);
GO

CREATE TABLE [SUPPLY]
(
    [SUPPLY_ID] int IDENTITY(1,1) PRIMARY KEY,
    [SUPPLY_DATE] datetime2 NOT NULL,
    [SUPPLIER_ID] int NOT NULL,
    [TOTAL_AMOUNT] decimal(18,2) NOT NULL,
    CONSTRAINT [FK_SUPPLY_SUPPLIER]
        FOREIGN KEY ([SUPPLIER_ID]) REFERENCES [SUPPLIER]([SUPPLIER_ID])
);
GO

CREATE TABLE [SUPPLY_ITEM]
(
    [SUPPLY_ITEM_ID] int IDENTITY(1,1) PRIMARY KEY,
    [SUPPLY_ID] int NOT NULL,
    [PRODUCT_ID] int NOT NULL,
    [QUANTITY] int NOT NULL,
    [UNIT_COST] decimal(18,2) NOT NULL,
    CONSTRAINT [FK_SUPPLY_ITEM_SUPPLY]
        FOREIGN KEY ([SUPPLY_ID]) REFERENCES [SUPPLY]([SUPPLY_ID]),
    CONSTRAINT [FK_SUPPLY_ITEM_PRODUCT]
        FOREIGN KEY ([PRODUCT_ID]) REFERENCES [PRODUCT]([PRODUCT_ID])
);
GO

CREATE TABLE [SALE]
(
    [SALE_ID] int IDENTITY(1,1) PRIMARY KEY,
    [SALE_DATE] datetime2 NOT NULL,
    [CUSTOMER_ID] int NULL,
    [SELLER_USER_ID] int NOT NULL,
    [TOTAL_AMOUNT] decimal(18,2) NOT NULL,
    CONSTRAINT [FK_SALE_CUSTOMER]
        FOREIGN KEY ([CUSTOMER_ID]) REFERENCES [CUSTOMER]([CUSTOMER_ID]),
    CONSTRAINT [FK_SALE_APP_USER]
        FOREIGN KEY ([SELLER_USER_ID]) REFERENCES [APP_USER]([USER_ID])
);
GO

CREATE TABLE [SALE_ITEM]
(
    [SALE_ITEM_ID] int IDENTITY(1,1) PRIMARY KEY,
    [SALE_ID] int NOT NULL,
    [PRODUCT_ID] int NOT NULL,
    [QUANTITY] int NOT NULL,
    [UNIT_PRICE] decimal(18,2) NOT NULL,
    [LINE_TOTAL] decimal(18,2) NOT NULL,
    CONSTRAINT [FK_SALE_ITEM_SALE]
        FOREIGN KEY ([SALE_ID]) REFERENCES [SALE]([SALE_ID]),
    CONSTRAINT [FK_SALE_ITEM_PRODUCT]
        FOREIGN KEY ([PRODUCT_ID]) REFERENCES [PRODUCT]([PRODUCT_ID])
);
GO

CREATE TABLE [WRITE_OFF]
(
    [WRITE_OFF_ID] int IDENTITY(1,1) PRIMARY KEY,
    [WRITE_OFF_DATE] datetime2 NOT NULL,
    [REASON] nvarchar(300) NOT NULL,
    [CREATED_BY_USER_ID] int NOT NULL,
    CONSTRAINT [FK_WRITE_OFF_APP_USER]
        FOREIGN KEY ([CREATED_BY_USER_ID]) REFERENCES [APP_USER]([USER_ID])
);
GO

CREATE TABLE [WRITE_OFF_ITEM]
(
    [WRITE_OFF_ITEM_ID] int IDENTITY(1,1) PRIMARY KEY,
    [WRITE_OFF_ID] int NOT NULL,
    [PRODUCT_ID] int NOT NULL,
    [QUANTITY] int NOT NULL,
    CONSTRAINT [FK_WRITE_OFF_ITEM_WRITE_OFF]
        FOREIGN KEY ([WRITE_OFF_ID]) REFERENCES [WRITE_OFF]([WRITE_OFF_ID]),
    CONSTRAINT [FK_WRITE_OFF_ITEM_PRODUCT]
        FOREIGN KEY ([PRODUCT_ID]) REFERENCES [PRODUCT]([PRODUCT_ID])
);
GO

CREATE TABLE [APP_CHANGE_LOG]
(
    [CHANGE_LOG_ID] int IDENTITY(1,1) PRIMARY KEY,
    [ENTITY_NAME] nvarchar(150) NOT NULL,
    [ENTITY_ID] nvarchar(150) NOT NULL,
    [ACTION_TYPE] nvarchar(30) NOT NULL,
    [OLD_VALUES] nvarchar(max) NULL,
    [NEW_VALUES] nvarchar(max) NULL,
    [CHANGED_AT] datetime2 NOT NULL DEFAULT SYSDATETIME(),
    [CHANGED_BY_USER_ID] int NULL,
    CONSTRAINT [FK_APP_CHANGE_LOG_APP_USER]
        FOREIGN KEY ([CHANGED_BY_USER_ID]) REFERENCES [APP_USER]([USER_ID])
);
GO
